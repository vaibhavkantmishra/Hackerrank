from itertools import product
k,m=map(int,input().split())
def sqaure(n):
    return int(n)**2
list1=[]
for i in range(k):
    list1.append(list(map(sqaure,input().split()[1:])))
Maximum=0
for ele in product(*list1):
    Sum=sum(ele)%m
    if Sum>Maximum:
        Maximum = Sum
        
print(Maximum)
